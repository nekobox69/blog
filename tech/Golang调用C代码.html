Golang调用C代码

2023年3月11日 · 阅读需 4 分钟

Golang调用C代码可以有三种方式：

  1. 内嵌C代码到.go文件
  2. C代码直接放在项目中，和Go代码混编
  3. 引用C编译好的库文件

windows平台​

Windows上是没有gcc程序的，要想运行C代码，需要先安装gcc。

推荐使用MSYS2

安装好后，运行以下命令：

pacman -S --needed base-devel mingw-w64-x86_64-toolchain make

配置Path环境变量：

D:\msys64\mingw64\bin

内嵌C代码​

main.go

package main

/*#include <stdio.h>

void say_hello_with_name(char * name){
    printf("hello %s\n", name);
}

unsigned long * array(unsigned short list[]) {
    static unsigned long out[4];
    for(int i=0; i<4;i++) {
        out[i] = list[i] + 1;
    }
    return &out[0];
}
*/
import "C"
import (
    "fmt"
    "unsafe"
)

func main() {
    C.say_hello_with_name(C.CString("oscar"))
    data := []uint16{1, 2, 3, 4}
    var pwd []uint32
    size := 4
    var param []C.ushort
    for _, v := range data {
        param = append(param, C.ushort(v))
    }
    out := C.array(&param[0])
    ptr := unsafe.Pointer(out)
    for i := 0; i < size; i++ {
        vptr := unsafe.Pointer(uintptr(ptr) + uintptr(i)*unsafe.Sizeof(C.ulong(1)))
        pwd = append(pwd, uint32(*(*C.ulong)(vptr)))
    }
    fmt.Println(pwd)
}

import "C"和注释之间不能用空行，否则会报错：could not determine kind of name for C.xxx

运行后：

[2 3 4 5]
hello oscar

混编​

文件目录：

├── demo.c
├── demo.h
└── main.go

demo.h

void say_hello_with_name(char * name);
unsigned long * array(unsigned short list[]);

demo.c

#include <stdio.h>
#include "demo.h"

void say_hello_with_name(char * name) {
    printf("hello %s\n", name);
}

unsigned long * array(unsigned short list[]) {
    static unsigned long out[4];
    for(int i=0; i<4;i++) {
        out[i] = list[i] + 1;
    }
    return &out[0];
}

库文件​

Windows​

编译库：

gcc -c demo.c
ar rcs libdemo.lib demo.o
gcc -fPIC -shared demo.c -o libdemo.dll

文件目录：

├── include
│   └── demo.h
├── linux
│   └── libdemo.lib
└── main.go

main.go

package main

/*

#cgo CFLAGS: -I./include

#cgo LDFLAGS: -L./linux -llibdemo

#include "demo.h"

*/
import "C"
import (
    "fmt"
    "unsafe"
)

func main() {
    C.say_hello_with_name(C.CString("oscar"))
